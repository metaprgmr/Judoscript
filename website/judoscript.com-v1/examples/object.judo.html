function my_comparator lhs, rhs {
  la = lhs.csv('.');
  ra = rhs.csv('.');
  lali = la.lastIndex();
  rali = ra.lastIndex();
  for i from 0 to min(lali, rali) {
    if la[i].int() < ra[i].int() { return -1; }
    if la[i].int() > ra[i].int() { return 1; }
  }
  if lali == rali { return 0; }
  return lali > rali;
}
