usage {
  author  = 'James Huang';
  created = '2001-11-12';
  lastMod = '2002-08-10';
  desc    = [[*
    Prints the ASCII characters for values between 32 and 255.
    Print the title fields aligned and repeated.
  *]];
}

cnt = 0;
println '   ', ' 0' @ (cnt++).fmtHex() {16};
println '   ', ' --' repeat 16;
for x from 2 to 15 {
  local base = x << 4;
  print x.fmtHex(), '0:';
  for y from 0 to 0x0f {
    local ch = base | y;
    print '  ', ch.chr();
  }
  println;
}
