usage {
  author  = 'James Huang';
  created = '2004-11-14';
  desc    = [[*
    Demonstrates the usage parsing JSP pages.

    You can use this to either transform JSPs, or simply
    use the JSP-like syntax to create custom templates.
    SGML tags are not affected and are treated as text.
  *]];
}

do $$local as jsp {
<%@page>: println $_;
<%>:      println $_;
<%=>:     println $_;
<%!>:     println $_;
<%-->:    println $_;
//:TEXT:    println $_;
}

EndScript =============================================
