text = '<head><title>Judo</title></head>';

m = text.matcher('(<title>)(.*)(</title>)');

if m.find() {
  println m.groupCount();
  println ' : ', m.group();
  println '0: ', m.group(0);
  println '1: ', m.group(1);
  println '2: ', m.group(2);
  println '3: ', m.group(3);
}
