  println '       protocol: ', b.protocol;
  println '         domain: ', b.domain;
  println '           host: ', b.host;
  println '           port: ', b.port;
  println '           path: ', b.path;
  println '      file_name: ', b.file_name;
  println '   query_string: ', b.query_string;
  println '            ref: ', b.ref;
  println 'constructed URL: ', b.transpose('UrlInfo');
}

class UrlInfo
{
  function toString() {
    ret = root + path;
    if query_string != null { ret += '?' + query_string; }
    if ref != null { ret += '#' + ref; }
    return ret;
  }
}
