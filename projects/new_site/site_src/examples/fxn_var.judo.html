function foo2 a, b { return a * b; }
function whichFoo f, a, b { return f(a,b); }

println 'whichFoo(&foo1, 2, 3) = ', whichFoo(&foo1, 2, 3);
println 'whichFoo(&foo2, 2, 3) = ', whichFoo(&foo2, 2, 3);

x = [ &foo1, &foo2 ];
println 'x = ', x;
println 'x[0](2, 3) = ', x[0](2, 3);
